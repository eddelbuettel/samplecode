Fitter function

myLinFit <- function(dat, offset = 1952) {
  theFit <- lm(lifeExp ~ I(year - offset), dat)
  setNames(coef(theFit), c("intercept", "slope"))
}

myLinFit(subset(gapminder, country == "Canada"))
## intercept     slope 
## 68.883846  0.218869
